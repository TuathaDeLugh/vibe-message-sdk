// 1. Install SDK
npm install fcm-clone-sdk

// 2. Import and initialize
import { initNotificationClient } from 'fcm-clone-sdk';

const client = initNotificationClient({
  baseUrl: 'http://localhost:3000/api',
  appId: 'your-app-id',
  publicKey: 'your-public-key'
});

// 3. Register callbacks
client.onMessage((payload) => {
  console.log('Foreground message:', payload);
  // Handle message when app is visible
});

client.onBackgroundMessage((payload) => {
  console.log('Background message:', payload);
  // Handle message when user clicks notification
});

client.onSilentMessage((data) => {
  console.log('Silent message:', data);
  // Handle silent data sync
});

// 4. Register device
await client.registerDevice({
  externalUserId: 'user-123',
  serviceWorkerPath: '/push-sw.js'
});

// 5. Unregister device
await client.unregisterDevice('user-123');
